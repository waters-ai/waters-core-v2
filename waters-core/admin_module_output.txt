use borsh::{BorshDeserialize, BorshSerialize};
use solana_program::{
    account_info::{next_account_info, AccountInfo},
    entrypoint::ProgramResult,
    msg,
    pubkey::Pubkey,
    sysvar::{rent::Rent, Sysvar},
};
use crate::{constants::*, error::WaterError};
use crate::instructions::utilits::pda::{
    derive_sol_fund_pda,
    get_root_pda_key,
};

// Константа для экстренного пополнения (3 вечных оплаты)
pub const EMERGENCY_REFILL_AMOUNT: u64 = 3 * 6_048_240; // ~0.018 SOL

#[derive(BorshSerialize, BorshDeserialize, Debug)]
pub struct EmergencyRefillParams {
    pub amount: u64, // Обычно EMERGENCY_REFILL_AMOUNT, но можно передать любую сумму
}

pub fn process_emergency_refill_sol_fund_shard(
    program_id: &Pubkey,
    accounts: &[AccountInfo],
    params: EmergencyRefillParams,
) -> ProgramResult {
    
    msg!("========================================================");
    msg!("🚨 [41] emergency_refill_sol_fund_shard START");
    msg!("========================================================");
    
    let account_iter = &mut accounts.iter();
    
    // 0. Root PDA
    let root_pda_account = next_account_info(account_iter)?;
    // 1. SOL Fund PDA (источник)
    let sol_fund = next_account_info(account_iter)?;
    // 2. Target shard (получатель)
    let shard = next_account_info(account_iter)?;

 
    // Получаем root_pda
    let root_pda = get_root_pda_key(program_id, root_pda_account)?;
    msg!("✅ Root PDA: {}", root_pda);

    // Проверяем SOL Fund PDA
    let (expected_sol_fund, _sol_bump) = derive_sol_fund_pda(program_id, &root_pda);
    if sol_fund.key != &expected_sol_fund {
        msg!("❌ Неверный SOL Fund PDA");
        return Err(WaterError::InvalidPDA.into());
    }

    // Проверяем, что SOL Fund принадлежит программе
    if sol_fund.owner != program_id {
        msg!("❌ SOL Fund не принадлежит программе");
        return Err(WaterError::IllegalOwner.into());
    }

    // Проверяем баланс
    let sol_fund_balance = **sol_fund.lamports.borrow();
    msg!("💰 Баланс SOL Fund: {} lamports", sol_fund_balance);
    msg!("💰 Требуется перевести: {} lamports", params.amount);

    if sol_fund_balance < params.amount {
        msg!("❌ Недостаточно средств в SOL Fund");
        return Err(WaterError::InsufficientFunds.into());
    }

    // Проверяем минимальную аренду для шарда после перевода
    let rent = Rent::get()?;
    let min_balance = rent.minimum_balance(shard.data_len());
    let new_shard_balance = (**shard.lamports.borrow()).saturating_add(params.amount);
    
    if new_shard_balance < min_balance {
        msg!("❌ Баланс шарда после перевода ниже минимальной аренды");
        return Err(WaterError::InsufficientFunds.into());
    }

    // Выполняем перевод
    msg!("💰 Переводим {} lamports из SOL Fund в шард {}", params.amount, shard.key);
    
    let fund_balance = **sol_fund.lamports.borrow();
    let shard_balance = **shard.lamports.borrow();
    
    **sol_fund.lamports.borrow_mut() = fund_balance
        .checked_sub(params.amount)
        .ok_or(WaterError::InsufficientFunds)?;
    
    **shard.lamports.borrow_mut() = shard_balance
        .checked_add(params.amount)
        .ok_or(WaterError::InvalidTransfer)?;

    msg!("✅ Экстренное пополнение выполнено успешно!");
    msg!("   Новый баланс SOL Fund: {} lamports", **sol_fund.lamports.borrow());
    msg!("   Новый баланс шарда: {} lamports", **shard.lamports.borrow());

    Ok(())
}===SEPARATOR===
// src/instructions/admin/redistribute_shard_balances.rs - ИСПРАВЛЕННАЯ ВЕРСИЯ

use borsh::{BorshDeserialize, BorshSerialize};
use solana_program::{
    account_info::{next_account_info, AccountInfo},
    entrypoint::ProgramResult,
    msg,
    pubkey::Pubkey,
    sysvar::{clock::Clock, Sysvar},
};
use crate::{
    constants::*,
    error::WaterError,
    state::{RentPdaMema, RentPdaShard},
    utils::validation_utils::validate_root_pda_v2,
};

#[derive(BorshSerialize, BorshDeserialize, Debug, Clone, Copy, PartialEq)]
#[borsh(use_discriminant = true)]  // ← ДОБАВЛЯЕМ ЭТОТ АТРИБУТ!
#[repr(u8)]
pub enum RedistributionMode {
    /// Только балансировка (balance_all_shards)
    BalanceOnly = 0,
    /// Только экстренные сигналы (handle_emergency_signals)
    EmergencyOnly = 1,
    /// Сначала экстренные, потом балансировка
    Full = 2,
}

#[derive(BorshSerialize, BorshDeserialize, Debug)]
pub struct RedistributeParams {
    pub mode: RedistributionMode,
}

pub fn process_redistribute_shard_balances(
    program_id: &Pubkey,
    accounts: &[AccountInfo],
    params: RedistributeParams,
) -> ProgramResult {
    
    msg!("========================================================");
    msg!("🔄 [40] redistribute_shard_balances START");
    msg!("========================================================");
    msg!("📋 Режим: {:?}", params.mode);
    
    let account_iter = &mut accounts.iter();
    
    // === Аккаунты ===
    let admin = next_account_info(account_iter)?;
    let rent_mema = next_account_info(account_iter)?;
    let root_pda_account = next_account_info(account_iter)?;
    
    // === Читаем все шарды (32 штуки) ===
    let mut shard_accounts = Vec::new();
    for _i in 0..TOTAL_RENT_SHARDS {
        let shard = next_account_info(account_iter)?;
        shard_accounts.push(shard);
    }
    
    msg!("📋 Аккаунтов: {}", 3 + TOTAL_RENT_SHARDS as usize);
    
    // === Валидация ===
    if !admin.is_signer {
        msg!("❌ Администратор не подписал");
        return Err(WaterError::Unauthorized.into());
    }
    
    let root_pda = validate_root_pda_v2(program_id, root_pda_account)?;
    
    // Проверяем rent_mema
    let (expected_mema, _) = Pubkey::find_program_address(
        &[RENTPDA_MEMA_SEED, root_pda.as_ref()],
        program_id,
    );
    if rent_mema.key != &expected_mema {
        msg!("❌ Неверный rent_mema PDA");
        return Err(WaterError::InvalidPDA.into());
    }
    
    // === Загружаем RentPdaMema ===
    let mut mema_data = RentPdaMema::try_from_slice(&rent_mema.data.borrow())
        .map_err(|_| WaterError::SerializationError)?;
    
    // Проверяем администратора
    mema_data.check_admin(admin.key)?;
    
    // === Загружаем все шарды ===
    let mut shards: Vec<RentPdaShard> = Vec::with_capacity(TOTAL_RENT_SHARDS as usize);
    for (i, shard_acc) in shard_accounts.iter().enumerate() {
        if shard_acc.data_is_empty() {
            msg!("⚠️ Шард {} не инициализирован", i);
            // Используем new() вместо empty()
            shards.push(RentPdaShard::new(
                *rent_mema.key,
                i as u8,
                0, // bump
            ));
            continue;
        }
        
        let shard = RentPdaShard::try_from_slice(&shard_acc.data.borrow())
            .map_err(|_| WaterError::SerializationError)?;
        
        if shard.rent_mema_pda != *rent_mema.key {
            msg!("❌ Шард {} привязан к другому rent_mema", i);
            return Err(WaterError::InvalidShardId.into());
        }
        
        shards.push(shard);
    }
    
    let now = Clock::get()?.unix_timestamp;
    
    // === Обновляем балансы шардов в mema_data ===
    for (i, shard) in shards.iter().enumerate() {
        mema_data.shard_balances[i] = shard.current_balance;
        mema_data.shard_priorities[i] = shard.priority;
    }
    
    // === ВЫЗЫВАЕМ ГОТОВЫЕ МЕТОДЫ ИЗ RENT_MEMA ===
    match params.mode {
        RedistributionMode::EmergencyOnly => {
            msg!("🚨 Только экстренные сигналы");
            
            if mema_data.shard_signals == 0 {
                msg!("✅ Нет активных сигналов");
                return Ok(());
            }
            
            let report = mema_data.handle_emergency_signals(&mut shards);
            
            msg!("📊 Экстренная обработка:");
            msg!("   Сигналов обработано: {}", report.signals_processed);
            msg!("   Сигналов истекло: {}", report.signals_expired);
            msg!("   Сигналов пропущено: {}", report.signals_skipped);
            msg!("   Шардов пополнено: {}", report.shards_refilled);
            msg!("   Сумма: {} lamports", report.amount_refilled);
        }
        
        RedistributionMode::BalanceOnly => {
            msg!("⚖️ Только балансировка");
            
            if now - mema_data.last_balance_check < REDISTRIBUTION_COOLDOWN {
                msg!("⏳ Кулдаун, пропускаем");
                return Ok(());
            }
            
            let report = mema_data.balance_all_shards(&mut shards);
            
            msg!("📊 Балансировка:");
            msg!("   Шардов пополнено: {}", report.shards_refilled);
            msg!("   Сумма: {} lamports", report.amount_refilled);
            msg!("   Сигналов обработано: {}", report.signals_processed);
            msg!("   Шардов проверено: {}", report.shards_checked);
            msg!("   Batch size: {}", report.batch_size);
            msg!("   Load level: {}", report.load_level);
        }
        
        RedistributionMode::Full => {
            msg!("🔁 Полный цикл: экстренные + балансировка");
            
            if mema_data.shard_signals != 0 {
                let emergency_report = mema_data.handle_emergency_signals(&mut shards);
                msg!("📊 Экстренные: {} сигналов, {} lamports",
                     emergency_report.signals_processed,
                     emergency_report.amount_refilled);
            }
            
            if now - mema_data.last_balance_check >= REDISTRIBUTION_COOLDOWN {
                let balance_report = mema_data.balance_all_shards(&mut shards);
                msg!("📊 Балансировка: {} шардов, {} lamports",
                     balance_report.shards_refilled,
                     balance_report.amount_refilled);
            } else {
                msg!("⏳ Балансировка в кулдауне, пропущена");
            }
        }
    }
    
    // === Обновляем данные шардов из mema_data ===
    for i in 0..shards.len() {
        shards[i].current_balance = mema_data.shard_balances[i];
        shards[i].priority = mema_data.shard_priorities[i];
    }
    
    // === Сохраняем обновленные шарды ===
    for (_i, (shard_acc, updated_shard)) in shard_accounts.iter().zip(shards.iter()).enumerate() {
        if !shard_acc.data_is_empty() {
            updated_shard.serialize(&mut *shard_acc.data.borrow_mut())?;
        }
    }
    
    // === Сохраняем обновленный rent_mema ===
    mema_data.last_stats_update = now;
    mema_data.serialize(&mut *rent_mema.data.borrow_mut())?;
    
    msg!("========================================================");
    msg!("✅ Инструкция 40 завершена");
    msg!("========================================================");
    
    Ok(())
}===SEPARATOR===

// src/instructions/admin/refill_funds.rs

use solana_program::{
    account_info::{next_account_info, AccountInfo},
    entrypoint::ProgramResult,
    msg,
    program::invoke,
    pubkey::Pubkey,
};
use crate::instructions::utils::get_associated_token_address;
use spl_token_2022_interface::instruction as token_instruction;
use spl_token_2022_interface::state::Mint as Token2022Mint;
use spl_token_2022_interface::extension::StateWithExtensions;

use crate::ProgramConfig;
use crate::state::config_trait::ConfigAccount;
use crate::constants::{BONUS_FUND_SEED, STAKING_FUND_SEED, FEE_VAULT_SEED};
use crate::error::WaterError;
use crate::utils::validation_utils::{validate_root_pda_v2, validate_signer, validate_token_account_mint}; 
use crate::utils::fund_manager::get_fund_pda;

pub fn process_admin_refill_funds(
    program_id: &Pubkey,
    accounts: &[AccountInfo],
    fund_type: u8,
    amount: u64,
) -> ProgramResult {
    let account_iter = &mut accounts.iter();

   // Получаем аккаунты
    let admin = next_account_info(account_iter)?; // подписант (должен быть authority)
    let program_config_account = next_account_info(account_iter)?; // ProgramConfig
    let immutable_seed_account = next_account_info(account_iter)?; // ImmutableSeedAccount
    let token_mint = next_account_info(account_iter)?; // токен mint
    let source_ata = next_account_info(account_iter)?; // откуда (admin ATA)
    let target_fund_ata = next_account_info(account_iter)?; // куда (фонд ATA)
    let token_program = next_account_info(account_iter)?;

    if !admin.is_signer {
        return Err(WaterError::Unauthorized.into());
    }

    // ✅ Централизованная проверка root_pda
    let root_pda = validate_root_pda_v2( program_id, immutable_seed_account,)?;


    // ✅ Усиленная проверка подписи администратора
    validate_signer(admin)?;

    // ✅ Десериализуем ProgramConfig для дополнительной проверки
    let program_config = ProgramConfig::get_config(program_config_account, program_id, &root_pda)?;

    // ✅ Проверка, что администратор совпадает с authority из конфига
    if admin.key != &program_config.authority {
        msg!("❌ Атака: администратор не совпадает с authority из конфига");
        msg!("  Ожидался: {}", program_config.authority);
        msg!("  Получен: {}", admin.key);
        return Err(WaterError::Unauthorized.into());
    }


    // ✅ Проверка валидности суммы пополнения
    if amount == 0 {
        msg!("❌ Сумма пополнения не может быть нулевой");
        return Err(WaterError::InvalidTransfer.into());
    }

    // Определяем целевой фонд
    let (fund_pda, _fund_bump) = match fund_type {
        0 => get_fund_pda(program_id, &root_pda, BONUS_FUND_SEED),
        1 => get_fund_pda(program_id, &root_pda, STAKING_FUND_SEED),
        2 => get_fund_pda(program_id, &root_pda, FEE_VAULT_SEED),
        _ => return Err(WaterError::InvalidConfig.into()),
    };

    msg!("🔍 Пополнение фонда:");
    msg!("  Тип: {}", match fund_type {
        0 => "Bonus Fund",
        1 => "Staking Fund",
        2 => "Fee Vault",
        _ => "Unknown",
    });
    msg!("  PDA фонда: {}", fund_pda);
    msg!("  Сумма: {}", amount);

    // ✅ Проверка: target_fund_ata — это ATA fund_pda
    let expected_ata = get_associated_token_address(&fund_pda, token_mint.key);
    if target_fund_ata.key != &expected_ata {
        msg!("❌ Атака: неверный target_fund_ata");
        msg!("  Ожидался: {}", expected_ata);
        msg!("  Получен: {}", target_fund_ata.key);
        return Err(WaterError::InvalidAccountData.into());
    }

    // ✅ Проверка, что source_ata принадлежит администратору
    validate_token_account_mint(source_ata, token_mint.key)?;

    // ✅ Проверка mint токен-аккаунтов
    validate_token_account_mint(target_fund_ata, token_mint.key)?;

    // ✅ Проверка баланса source_ata
    let source_data = source_ata.data.borrow();
    let source_token = StateWithExtensions::<spl_token_2022_interface::state::Account>::unpack(&source_data)
        .map_err(|_| {
            msg!("❌ Не удалось распаковать source_ata");
            WaterError::InvalidAccountData
        })?;

    let source_balance = source_token.base.amount;
    if source_balance < amount {
        msg!("❌ Недостаточно средств для пополнения");
        msg!("  Баланс source: {}", source_balance);
        msg!("  Запрошено: {}", amount);
        return Err(WaterError::InsufficientBalance.into());
    }

    // Получаем decimals
    let mint_data = token_mint.try_borrow_data()?;
    let mint = StateWithExtensions::<Token2022Mint>::unpack(&mint_data)
        .map_err(|_| WaterError::InvalidAccountData)?;
    let decimals = mint.base.decimals;

    // ✅ Проверяем, что source_ata действительно принадлежит администратору
    if source_token.base.owner != *admin.key {
        msg!("❌ Атака: source_ata не принадлежит администратору");
        msg!("  Владелец source_ata: {}", source_token.base.owner);
        msg!("  Администратор: {}", admin.key);
        return Err(WaterError::Unauthorized.into());
    }


    // Создаём инструкцию
    let transfer_ix = token_instruction::transfer_checked(
        token_program.key,
        source_ata.key,     // source: ATA администратора
        token_mint.key,
        target_fund_ata.key, // destination: целевой фонд
        admin.key,        // authority
        &[],              // multi_signers
        amount,
        decimals,
    ).map_err(|_| WaterError::InvalidTransfer)?;

    invoke(
        &transfer_ix,
        &[
            source_ata.clone(),
            token_mint.clone(),
            target_fund_ata.clone(),
            admin.clone(),           // authority (администратор)
            token_program.clone(),
        ],
    )?;

    msg!("✅ Фонд успешно пополнен администратором!");
    msg!("  Администратор: {}", admin.key);
    msg!("  Из: {} (баланс до: {}, после: {})",

    accounts: &[AccountInfo],
    params: RefillShardParams,

    msg!("💰 Сумма перевода: {} {}", params.amount,

fn process_token_refill(
    program_id: &Pubkey,
    account_iter: &mut std::slice::Iter<'_, AccountInfo<'_>>,
    amount: u64,
) -> ProgramResult {
    // 0. Root PDA
    let root_pda_account = next_account_info(account_iter)?;
    // 1. SOL Fund PDA (источник токенов)
    let sol_fund = next_account_info(account_iter)?;
    // 2. SOL Fund ATA (токеный счет фонда)
    let sol_fund_ata = next_account_info(account_iter)?;
    // 3. Target shard
    let shard = next_account_info(account_iter)?;
    // 4. Shard ATA (получатель токенов)
    let shard_ata = next_account_info(account_iter)?;
    // 5. Token Mint
    let token_mint = next_account_info(account_iter)?;
    // 6. Token Program
    let token_program = next_account_info(account_iter)?;

    msg!("========================================================");
    msg!("🚀 process_token_refill через SOL Fund START");
    msg!("========================================================");
    
    msg!("📋 Полученные аккаунты:");
    msg!("   0 root_pda: {} (owner: {})", root_pda_account.key, root_pda_account.owner);
    msg!("   1 sol_fund: {} (owner: {})", sol_fund.key, sol_fund.owner);
    msg!("   2 sol_fund_ata: {} (owner: {})", sol_fund_ata.key, sol_fund_ata.owner);
    msg!("   3 shard: {} (owner: {})", shard.key, shard.owner);
    msg!("   4 shard_ata: {} (owner: {})", shard_ata.key, shard_ata.owner);
    msg!("   5 token_mint: {} (owner: {})", token_mint.key, token_mint.owner);
    msg!("   6 token_program: {} (owner: {})", token_program.key, token_program.owner);

    // Получаем root_pda
    let root_pda = get_root_pda_key(program_id, root_pda_account)?;
    msg!("✅ Root PDA: {}", root_pda);

    // Получаем bump для SOL Fund PDA
    let (expected_sol_fund, bump) = derive_sol_fund_pda(program_id, &root_pda);
    
    if sol_fund.key != &expected_sol_fund {
        msg!("❌ Неверный SOL Fund PDA");
        msg!("   Ожидался: {}", expected_sol_fund);
        msg!("   Получен: {}", sol_fund.key);
        return Err(WaterError::InvalidPDA.into());
    }
    msg!("✅ SOL Fund PDA верный, bump: {}", bump);
    msg!("   Владелец SOL Fund: {}", sol_fund.owner);
    msg!("   Ожидаемый владелец (program_id): {}", program_id);
    
    if sol_fund.owner != program_id {
        msg!("❌ SOL Fund не принадлежит программе!");
        msg!("   Владелец: {}, Ожидался: {}", sol_fund.owner, program_id);
        return Err(WaterError::InvalidAccountData.into());
    }

    // Проверяем ATA SOL Fund
    let expected_ata = get_associated_token_address_with_program(
        sol_fund.key,
        token_mint.key,
        token_program.key
    );

    if sol_fund_ata.key != &expected_ata {
        msg!("❌ Неверный SOL Fund ATA");
        msg!("   Ожидался: {}", expected_ata);
        msg!("   Получен: {}", sol_fund_ata.key);
        return Err(WaterError::InvalidAccountData.into());
    }
    msg!("✅ SOL Fund ATA верный");

    // Получаем decimals из mint
    let mint_data = token_mint.try_borrow_data()?;
    let mint = StateWithExtensions::<Mint>::unpack(&mint_data)
        .map_err(|_| WaterError::InvalidAccountData)?;
    let decimals = mint.base.decimals;

    msg!("💰 Переводим {} токенов (decimals: {}) из SOL Fund в шард {}", 
         amount, decimals, shard.key);
    msg!("   From ATA: {} (balance before check)", sol_fund_ata.key);
    msg!("   To ATA: {} (balance before check)", shard_ata.key);

    let transfer_ix = token_instruction::transfer_checked(
        token_program.key,
        sol_fund_ata.key,
        token_mint.key,
        shard_ata.key,
        sol_fund.key,
        &[],
        amount,
        decimals,
    ).map_err(|_| WaterError::InvalidTransfer)?;

    let seeds = &[
        SOL_FUND_SEED,
        root_pda.as_ref(),
        &[bump]
    ];
    let signer_seeds = &[&seeds[..]];

    msg!("🔑 Seeds для подписи: {:?}", seeds);
    msg!("📝 Вызов invoke_signed для перевода токенов...");

    invoke_signed(
        &transfer_ix,
        &[
            sol_fund_ata.clone(),
            token_mint.clone(),
            shard_ata.clone(),
            sol_fund.clone(),
            token_program.clone(),
        ],
        signer_seeds,
    )?;

    msg!("✅ Перевод токенов через SOL Fund выполнен успешно");
    Ok(())
}

/// Raw AdminAuth check — reads only admin_1 pubkey + solo_hash + blocked_until from raw bytes
fn check_auth_raw(data: &[u8], signer: &Pubkey, solo_hash: &[u8; 32]) -> Result<(), WaterError> {
    if data.len() < 72 { return Err(WaterError::Unauthorized); }
    let pk = Pubkey::new_from_array(data[0..32].try_into().unwrap());
    if *signer != pk { return Err(WaterError::Unauthorized); }
    let stored: [u8; 32] = data[32..64].try_into().unwrap();
    if solo_hash != &stored { return Err(WaterError::InvalidSoloPassword); }
    let blocked = i64::from_le_bytes(data[64..72].try_into().unwrap());
    let now = solana_program::clock::Clock::get().map_err(|_| WaterError::InvalidTimestamp)?.unix_timestamp;
    if blocked > now { return Err(WaterError::Unauthorized); }
    Ok(())
}

pub fn process_withdraw_rent(

    accounts: &[AccountInfo],
    amount: u64,
) -> ProgramResult {

    let admin = next_account_info(account_iter)?;
    let rentpda_mema = next_account_info(account_iter)?;

    if !admin.is_signer {
        return Err(ProgramError::MissingRequiredSignature);
    }

    let (expected_pda, _bump) = Pubkey::find_program_address(&[b"rentpda_mema"], program_id);
    if rentpda_mema.key != &expected_pda || rentpda_mema.owner != program_id {
        return Err(ProgramError::IllegalOwner);
    }

    // ✅ Проверка баланса rent_pda перед списанием
    if rentpda_mema.lamports() < amount {
        msg!("❌ Недостаточно средств в rent_pda для вывода");
        msg!("  Запрошено: {} lamports", amount);
        msg!("  Доступно: {} lamports", rentpda_mema.lamports());
        return Err(ProgramError::InsufficientFunds);
    }

    **rentpda_mema.try_borrow_mut_lamports()? -= amount;
    **admin.try_borrow_mut_lamports()? += amount;

    Ok(())
}

    mema.serialize(&mut *rent_mema.data.borrow_mut())
        .map_err(|_| {
            msg!("❌ Ошибка сериализации RentPdaMema");
            WaterError::SerializationError
        })?;
    
    // ✅ Переводим SOL из RentPdaMema в treasury
    msg!("💸 Перевод {} lamports в treasury {}", params.amount, treasury.key);
    
    let transfer_ix = system_instruction::transfer(
        rent_mema.key,
        treasury.key,
        params.amount,
    );
    
    // Подписываем как RentPdaMema PDA
    let seeds = &[RENTPDA_MEMA_SEED, root_pda.as_ref(), &[mema_bump]];
    
    invoke_signed(
        &transfer_ix,
        &[
            rent_mema.clone(),
            treasury.clone(),
            system_program.clone(),
        ],
        &[seeds],
    )?;
    
    msg!("========================================================");
    msg!("🎉 ВЫВОД СРЕДСТВ УСПЕШНО ВЫПОЛНЕН");
    msg!("========================================================");
    msg!("   Сумма: {} lamports ({:.3} SOL)", 
         params.amount, params.amount as f64 / 1_000_000_000.0);
    msg!("   Новый баланс RentPdaMema: {} lamports", mema.current_balance);
    msg!("   Доступно для вывода: {} lamports", mema.available_balance);
    msg!("   Получатель: {}", treasury.key);
    msg!("========================================================");
    
    Ok(())
}
